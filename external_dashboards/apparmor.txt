#include <tunables/global>

profile external_dashboards flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  network,
  signal (send) set=(kill,term,int,hup,cont),

  # Node.js runtime (official node:24-alpine image places node under
  # /usr/local/bin). /usr/bin is covered as a fallback for future rebasing.
  /usr/local/bin/node ix,
  /usr/local/bin/npm ix,
  /usr/bin/node ix,
  /usr/local/lib/node_modules/** r,
  /usr/lib/node_modules/** r,

  # Alpine core binaries + busybox shell (healthcheck, entrypoint, etc.).
  /bin/** ix,
  /usr/bin/** ix,
  /sbin/** ix,
  /usr/sbin/** ix,
  /etc/** r,

  # Application tree.
  /app/** rix,
  /tmp/** rwk,

  # Persistent add-on state (options.json, secrets.json).
  /data/** rwk,

  # User config dir: SQLite database + uploaded assets.
  /config/** rwk,

  # Legacy / HA-base paths retained as no-ops — harmless under
  # node:24-alpine but keep the profile forward-compatible if we ever
  # rebase on an HA base image that uses s6-overlay.
  /init ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,
  /usr/lib/bashio/** ix,

  # Deny the dangerous stuff explicitly so a future refactor can't
  # accidentally widen access.
  deny /root/** rwkl,
  deny /home/** rwkl,
  deny /etc/shadow r,
  deny /proc/*/mem r,
  deny /proc/sys/** w,
  deny /sys/** w,
  deny capability mac_admin,
  deny capability mac_override,
  deny capability sys_admin,
  deny capability sys_module,
  deny capability sys_ptrace,
  deny capability sys_rawio,
  deny capability dac_read_search,
}
